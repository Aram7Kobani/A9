maakgebruiker.sh
------------------
#!/bin/bash

#Dit script maakt een gebruiker aan met wachtwoord

echo "Voer de gebruikersnaam in die je wilt aanmaken:"
read GEBRUIKER

sudo useradd -m "$GEBRUIKER"

echo "Voer nu een wachtwoord in voor gebruiker $GEBRUIKER:"
sudo passwd "$GEBRUIKER"

echo "Gebruiker $GEBRUIKER is aangemaakt."

verwijdergebruiker.sh
----------------------
#!/bin/bash

#Dit script verwijdert een gebruiker en zijn homedirectory

echo "Welke gebruiker wil je verwijderen?"
read GEBRUIKER

sudo userdel -r "$GEBRUIKER"

echo "Gebruiker $GEBRUIKER is verwijderd."
